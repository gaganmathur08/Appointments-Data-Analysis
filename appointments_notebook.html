#1) Clean datetime. convert it into datetime format.

def parse_date(date):
    if date == '':
        return None
    else:
        return dt.strptime(date,'%Y-%m-%dT%H:%M:%SZ')

#For calculating pearsons r value between age and diabetes.
def find_correlation_r(x,y):
    std_x = (x-x.mean())/x.std(ddof=0)
    std_y = (y-y.mean())/y.std(ddof=0)
    
    return (std_x * std_y).mean()

print(find_correlation_r(unique_appointment_df['Age'],unique_appointment_df['Diabetes']))

    elif age <=19:
        return 'Adolscents'

The dataset given does not contains city,state or country data otherwise we could have identified which cities or states are having more number of patients.
